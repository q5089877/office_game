# 流體縮放與自適應畫布實施方案

## 概述
本方案旨在實現遊戲畫布的流體縮放和自適應功能，確保在不同裝置和螢幕尺寸上都能提供最佳的視覺體驗和互動性。

## 當前問題分析

### 現有實現 (`App.tsx` 第37-49行)
```typescript
React.useEffect(() => {

    const availableWidth = window.innerWidth - CANVAS_CONFIG.UI.SIDEBAR_WIDTH;

```

### 問題點
1. **固定UI尺寸**：側邊欄320px、底部區域220px固定
2. **縮放範圍限制**：最小0.5，最大1.2，不適合小螢幕
3. **緩衝係數固定**：0.9固定，未考慮不同裝置
4. **未考慮裝置方向**：橫向/縱向切換未優化
5. **性能問題**：無防抖動處理

## 改進方案

### 1. 智慧縮放計算器 (`src/utils/CanvasScaleManager.ts`)

#### 核心算法
```typescript
export class CanvasScaleManager {
  /**
   * 計算最佳縮放比例
   * @param availableWidth 可用寬度
   * @param availableHeight 可用高度
   * @param baseWidth 基礎寬度
   * @param baseHeight 基礎高度
   * @param layoutMode 佈局模式
   * @param options 配置選項
   */
  static calculateOptimalScale(
    availableWidth: number,
    availableHeight: number,
    baseWidth: number,
    baseHeight: number,
    layoutMode: LayoutMode,
    options: ScaleOptions = {}
  ): ScaleResult {
    const {
      minScale = 0.25,
      maxScale = 2.0,
      paddingFactor = 0.9,
      aspectRatioAdjust = 1.0,
      preserveAspectRatio = true
    } = options;

    // 計算基礎縮放比例
    const scaleX = availableWidth / baseWidth;
    const scaleY = availableHeight / baseHeight;

    let scale: number;

    if (preserveAspectRatio) {
      // 保持寬高比，取最小比例
      scale = Math.min(scaleX, scaleY) * paddingFactor;
    } else {
      // 不保持寬高比，獨立縮放
      scale = Math.min(scaleX, scaleY) * paddingFactor;
    }

    // 應用寬高比調整
    scale *= aspectRatioAdjust;

    // 限制在最小和最大範圍內
    scale = Math.max(minScale, Math.min(maxScale, scale));

    // 計算實際尺寸
    const width = baseWidth * scale;
    const height = baseHeight * scale;

    // 計算位置（置中）
    const offsetX = (availableWidth - width) / 2;
    const offsetY = (availableHeight - height) / 2;

    return {
      scale,
      width,
      height,
      offsetX,
      offsetY,
      fitsScreen: scale >= minScale && scale <= maxScale
    };
  }

  /**
   * 根據裝置類型獲取縮放策略
   */
  static getScaleStrategy(layoutMode: LayoutMode): ScaleStrategy {
    const strategies: Record<LayoutMode, ScaleStrategy> = {
      'mobile': {
        minScale: 0.25,
        maxScale: 0.7,
        paddingFactor: 0.8,
        aspectRatioAdjust: 0.9,
        debounceMs: 200
      },
      'tablet': {
        minScale: 0.35,
        maxScale: 0.9,
        paddingFactor: 0.85,
        aspectRatioAdjust: 0.95,
        debounceMs: 150
      },
      'desktop': {
        minScale: 0.5,
        maxScale: 1.2,
        paddingFactor: 0.9,
        aspectRatioAdjust: 1.0,
        debounceMs: 100
      },
      'wide': {
        minScale: 0.6,
        maxScale: 1.5,
        paddingFactor: 0.95,
        aspectRatioAdjust: 1.05,
        debounceMs: 100
      },
      'ultra-wide': {
        minScale: 0.7,
        maxScale: 2.0,
        paddingFactor: 0.98,
        aspectRatioAdjust: 1.1,
        debounceMs: 100
      }
    };

    return strategies[layoutMode];
  }

  /**
   * 處理視窗大小變化
   */
  static setupResizeHandler(
    callback: (result: ScaleResult) => void,
    baseWidth: number,
    baseHeight: number,
    layoutMode: LayoutMode,
    getAvailableSpace: () => { width: number; height: number }
  ): () => void {
    const strategy = this.getScaleStrategy(layoutMode);

    const calculateAndNotify = () => {
      const { width: availableWidth, height: availableHeight } = getAvailableSpace();
      const result = this.calculateOptimalScale(
        availableWidth,
        availableHeight,
        baseWidth,
        baseHeight,
        layoutMode,
        strategy
      );
      callback(result);
    };

    // 防抖動處理
    const debouncedCalculate = debounce(calculateAndNotify, strategy.debounceMs);

    // 監聽事件
    window.addEventListener('resize', debouncedCalculate);
    window.addEventListener('orientationchange', debouncedCalculate);

    // 初始計算
    calculateAndNotify();

    // 返回清理函數
    return () => {
      window.removeEventListener('resize', debouncedCalculate);
      window.removeEventListener('orientationchange', debouncedCalculate);
    };
  }
}

// 類型定義
interface ScaleOptions {
  minScale?: number;
  maxScale?: number;
  paddingFactor?: number;
  aspectRatioAdjust?: number;
  preserveAspectRatio?: boolean;
}

interface ScaleStrategy extends ScaleOptions {
  debounceMs: number;
}

interface ScaleResult {
  scale: number;
  width: number;
  height: number;
  offsetX: number;
  offsetY: number;
  fitsScreen: boolean;
}
```

### 2. 可用空間計算器 (`src/utils/AvailableSpaceCalculator.ts`)

```typescript
export class AvailableSpaceCalculator {
  /**
   * 計算畫布可用空間
   */
  static calculateAvailableSpace(
    windowWidth: number,
    windowHeight: number,
    layoutMode: LayoutMode,
    sidebarWidth: string,
    bottomAreaHeight: string
  ): { width: number; height: number } {
    // 解析CSS單位
    const sidebarWidthPx = this.parseCssLength(sidebarWidth, windowWidth);
    const bottomHeightPx = this.parseCssLength(bottomAreaHeight, windowHeight);

    let availableWidth = windowWidth;
    let availableHeight = windowHeight;

    // 根據佈局模式調整
    switch (layoutMode) {
      case 'mobile':
        // 手機：垂直堆疊，全寬度
        availableHeight -= bottomHeightPx;
        break;

      case 'tablet':
        // 平板：側邊欄可折疊，假設展開
        availableWidth -= sidebarWidthPx;
        availableHeight -= bottomHeightPx;
        break;

      case 'desktop':
      case 'wide':
      case 'ultra-wide':
        // 桌面：固定側邊欄
        availableWidth -= sidebarWidthPx;
        availableHeight -= bottomHeightPx;
        break;
    }

    // 確保最小尺寸
    availableWidth = Math.max(100, availableWidth);
    availableHeight = Math.max(100, availableHeight);

    return { width: availableWidth, height: availableHeight };
  }

  /**
   * 解析CSS長度單位
   */
  private static parseCssLength(value: string, reference: number): number {
    if (value.endsWith('px')) {
      return parseFloat(value);
    } else if (value.endsWith('%')) {
      return (parseFloat(value) / 100) * reference;
    } else if (value === '100%') {
      return reference;
    } else {
      // 預設為px
      return parseFloat(value) || 0;
    }
  }

  /**
   * 計算響應式邊距
   */
  static getResponsivePadding(layoutMode: LayoutMode): {
    top: string;
    right: string;
    bottom: string;
    left: string;
  } {
    const paddings = {
      'mobile': { top: '0.5rem', right: '0.5rem', bottom: '0.5rem', left: '0.5rem' },
      'tablet': { top: '1rem', right: '1rem', bottom: '1rem', left: '1rem' },
      'desktop': { top: '1.5rem', right: '1.5rem', bottom: '1.5rem', left: '1.5rem' },
      'wide': { top: '2rem', right: '2rem', bottom: '2rem', left: '2rem' },
      'ultra-wide': { top: '2.5rem', right: '2.5rem', bottom: '2.5rem', left: '2.5rem' }
    };

    return paddings[layoutMode];
  }
}
```

### 3. 畫布組件改進 (`src/components/Canvas/OfficeCanvas.tsx`)

#### 改進點
1. **動態尺寸**：根據計算結果調整畫布尺寸
2. **觸控支援**：添加手勢縮放
3. **性能優化**：減少不必要的重新渲染

```typescript
// 改進後的OfficeCanvas組件
interface EnhancedOfficeCanvasProps extends OfficeCanvasProps {
  scaleResult: ScaleResult;
  onScaleChange?: (scale: number) => void;
  enableTouchZoom?: boolean;
}

const EnhancedOfficeCanvas: React.FC<EnhancedOfficeCanvasProps> = ({
  gameState,
  scaleResult,
  onScaleChange,
  enableTouchZoom = true,
  ...props
}) => {
  const containerRef = useRef<HTMLDivElement>(null);
  const [localScale, setLocalScale] = useState(scaleResult.scale);

  // 合併縮放比例
  const effectiveScale = scaleResult.scale * localScale;

  // 設置觸控縮放
  useEffect(() => {
    if (!containerRef.current || !enableTouchZoom) return;

    const cleanup = TouchInteraction.setupPinchToZoom(
      containerRef.current,
      (scale) => {
        setLocalScale(scale);
        onScaleChange?.(scale);
      }
    );

    return cleanup;
  }, [enableTouchZoom, onScaleChange]);

  // 重置本地縮放當外部縮放變化時
  useEffect(() => {
    setLocalScale(1);
  }, [scaleResult.scale]);

  return (
    <div
      ref={containerRef}
      className="bg-white rounded-[32px] shadow-2xl border border-slate-200/60 overflow-hidden relative flex items-center justify-center"
      style={{
        width: `${scaleResult.width}px`,
        height: `${scaleResult.height}px`,
        marginLeft: `${scaleResult.offsetX}px`,
        marginTop: `${scaleResult.offsetY}px`,
        touchAction: enableTouchZoom ? 'none' : 'auto'
      }}
    >
      <Stage
        width={scaleResult.width}
        height={scaleResult.height}
        scaleX={effectiveScale}
        scaleY={effectiveScale}
      >
        {/* 原有圖層內容 */}
        <Layer>
          {/* 畫布背景 */}
          <Rect
            width={CANVAS_CONFIG.BASE_WIDTH}
            height={CANVAS_CONFIG.BASE_HEIGHT}
            fill="#fff"
          />

          {/* 其他內容... */}
        </Layer>
      </Stage>

      {/* 縮放指示器（僅在觸控縮放時顯示） */}
      {enableTouchZoom && localScale !== 1 && (
        <div className="absolute bottom-4 right-4 bg-black/70 text-white text-xs px-2 py-1 rounded">
          {Math.round(localScale * 100)}%
        </div>
      )}
    </div>
  );
};
```

// 在App組件中
export default function App() {
  const [scaleResult, setScaleResult] = useState<ScaleResult>({
    scale: 1,
    width: CANVAS_CONFIG.BASE_WIDTH,
    height: CANVAS_CONFIG.BASE_HEIGHT,
    offsetX: 0,
    offsetY: 0,
    fitsScreen: true
  });

  const { deviceInfo } = useResponsiveLayout(
    CANVAS_CONFIG.BASE_WIDTH,
    CANVAS_CONFIG.BASE_HEIGHT
  );

  // 計算可用空間的回調函數
  const getAvailableSpace = useCallback(() => {
    const sidebarWidth = getSidebarWidth(deviceInfo.layoutMode);
    const bottomHeight = getBottomAreaHeight(deviceInfo.layoutMode);

    return AvailableSpaceCalculator.calculateAvailableSpace(
      window.innerWidth,
      window.innerHeight,
      deviceInfo.layoutMode,
      sidebarWidth,
      bottomHeight
    );
  }, [deviceInfo.layoutMode]);

  // 設置縮放處理器
  useEffect(() => {
    const cleanup = CanvasScaleManager.setupResizeHandler(
      (result) => {
        setScaleResult(result);
      },
      CANVAS_CONFIG.BASE_WIDTH,
      CANVAS_CONFIG.BASE_HEIGHT,
      deviceInfo.layoutMode,
      getAvailableSpace
    );

    return cleanup;
  }, [deviceInfo.layoutMode, getAvailableSpace]);

  // 處理畫布縮放變化
  const handleCanvasScaleChange = useCallback((scale: number) => {
    console.log('Canvas scale changed:', scale);
    // 可以保存使用者偏好或觸發其他操作
  }, []);

  return (
    <div className={`h-screen w-screen ${tw.bg.light} ${tw.text.primary} font-sans overflow-hidden`}>
      {/* 根據佈局模式渲染不同的佈局 */}
      {deviceInfo.layoutMode === 'mobile' ? (
        <MobileLayout
          sidebar={<Sidebar {...sidebarProps} />}
          canvas={
            <EnhancedOfficeCanvas

          bottomArea={<BottomCardArea {...bottomProps} />}
        />
      ) : deviceInfo.layoutMode === 'tablet' ? (
        <TabletLayout
          sidebar={<Sidebar {...sidebarProps} />}
          canvas={
            <EnhancedOfficeCanvas
              gameState={gameState}
              scaleResult={scaleResult}
              onScaleChange={handleCanvasScaleChange}
              enableTouchZoom={deviceInfo.isTouchDevice}
              {...canvasProps}
            />
          }
          bottomArea={<BottomCardArea {...bottomProps} />}
        />
      ) : (
        <DesktopLayout
          sidebar={<Sidebar {...sidebarProps} />}
          canvas={
            <EnhancedOfficeCanvas
              gameState={gameState}
              scaleResult={scaleResult}
              onScaleChange={handleCanvasScaleChange}
              enableTouchZoom={deviceInfo.isTouchDevice}
              {...canvasProps}
            />
          }
          bottomArea={<BottomCardArea {...bottomProps} />}
        />
      )}
    </div>
  );
}
