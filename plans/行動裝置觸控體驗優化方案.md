# 行動裝置觸控體驗優化方案

## 概述
本方案旨在為遊戲提供最佳的行動裝置觸控體驗，確保所有互動元素都適合手指操作，並提供直觀的手勢控制。

## 設計原則

### 1. 觸控優先設計 (Touch-First Design)
- 所有互動元素最小尺寸為44x44px（WCAG標準）
- 充足的間距防止誤觸
- 視覺回饋和觸覺回饋（如果支援）

### 2. 手勢直觀性 (Gesture Intuitiveness)
- 符合使用者預期的手勢
- 提供手勢提示和教學
- 避免手勢衝突

### 3. 性能優化 (Performance Optimization)
- 60fps流暢動畫
- 減少觸控延遲
- 高效的事件處理

## 具體實施方案

### 1. 觸控目標優化系統

#### 觸控目標檢查器 (`src/utils/TouchTargetOptimizer.ts`)
```typescript
export class TouchTargetOptimizer {
  // 最小觸控目標尺寸（符合WCAG標準）
  static readonly MIN_TOUCH_SIZE = 44; // px
  static readonly MIN_TOUCH_MARGIN = 8; // px

  /**
   * 優化單個元素的觸控目標
   */
  static optimizeElement(element: HTMLElement): void {
    const rect = element.getBoundingClientRect();
    const currentWidth = rect.width;
    const currentHeight = rect.height;

    // 檢查是否需要優化
    if (currentWidth >= this.MIN_TOUCH_SIZE && currentHeight >= this.MIN_TOUCH_SIZE) {
      return; // 已經符合標準
    }

    // 計算需要的擴展
    const scaleX = Math.max(1, this.MIN_TOUCH_SIZE / currentWidth);
    const scaleY = Math.max(1, this.MIN_TOUCH_SIZE / currentHeight);
    const scale = Math.max(scaleX, scaleY);

    // 應用優化
    this.applyTouchOptimization(element, scale);
  }

  /**
   * 批量優化觸控目標
   */
  static optimizeAll(selector: string = '[data-touch-target]'): void {
    const elements = document.querySelectorAll<HTMLElement>(selector);
    elements.forEach(element => this.optimizeElement(element));
  }

  /**
   * 應用觸控優化
   */
  private static applyTouchOptimization(element: HTMLElement, scale: number): void {
    // 添加CSS類別
    element.classList.add('touch-optimized');

    // 設置觸控樣式
    element.style.minWidth = `${this.MIN_TOUCH_SIZE}px`;
    element.style.minHeight = `${this.MIN_TOUCH_SIZE}px`;
    element.style.padding = `${this.MIN_TOUCH_MARGIN}px`;

    // 如果元素太小，使用transform放大
    if (scale > 1.1) {
      element.style.transform = `scale(${scale})`;
      element.style.transformOrigin = 'center';
      element.style.zIndex = '10'; // 確保在頂層
    }

    // 添加觸控回饋
    element.addEventListener('touchstart', this.handleTouchStart, { passive: true });
    element.addEventListener('touchend', this.handleTouchEnd, { passive: true });
  }

  /**
   * 觸控開始回饋
   */
  private static handleTouchStart(e: TouchEvent): void {
    const target = e.currentTarget as HTMLElement;
    target.classList.add('touch-active');

    // 觸覺回饋（如果支援）
    if (navigator.vibrate) {
      navigator.vibrate(10);
    }
  }

  /**
   * 觸控結束回饋
   */
  private static handleTouchEnd(e: TouchEvent): void {
    const target = e.currentTarget as HTMLElement;
    target.classList.remove('touch-active');
  }

  /**
   * 創建觸控友好的按鈕
   */
  static createTouchButton(
    text: string,
    onClick: () => void,
    options: TouchButtonOptions = {}
  ): HTMLButtonElement {
    const button = document.createElement('button');
    button.textContent = text;
    button.className = 'touch-button';

    // 設置觸控樣式
    button.style.minWidth = `${this.MIN_TOUCH_SIZE}px`;
    button.style.minHeight = `${this.MIN_TOUCH_SIZE}px`;
    button.style.padding = `${this.MIN_TOUCH_MARGIN * 2}px`;
    button.style.fontSize = '16px'; // 最小可讀字體大小
    button.style.borderRadius = '8px'; // 圓角更適合觸控

    // 添加事件監聽器
    button.addEventListener('click', onClick);
    button.addEventListener('touchstart', this.handleTouchStart, { passive: true });
    button.addEventListener('touchend', this.handleTouchEnd, { passive: true });

    return button;
  }
}

// 類型定義
interface TouchButtonOptions {
  variant?: 'primary' | 'secondary' | 'danger';
  size?: 'small' | 'medium' | 'large';
  disabled?: boolean;
}
```

### 2. 手勢控制系統

#### 手勢管理器 (`src/utils/GestureManager.ts`)
```typescript
export class GestureManager {
  // 手勢類型
  static readonly GESTURE_TYPES = {
    TAP: 'tap',
    DOUBLE_TAP: 'double-tap',
    LONG_PRESS: 'long-press',
    SWIPE_LEFT: 'swipe-left',
    SWIPE_RIGHT: 'swipe-right',
    SWIPE_UP: 'swipe-up',
    SWIPE_DOWN: 'swipe-down',
    PINCH_IN: 'pinch-in',
    PINCH_OUT: 'pinch-out'
  } as const;

  // 配置
  private config = {
    tapThreshold: 300, // ms
    doubleTapThreshold: 500, // ms
    longPressThreshold: 800, // ms
    swipeThreshold: 50, // px
    swipeVelocityThreshold: 0.3, // px/ms
    pinchThreshold: 0.1 // 比例變化
  };

  // 狀態
  private touchStartTime = 0;
  private touchStartX = 0;
  private touchStartY = 0;
  private lastTapTime = 0;
  private initialDistance = 0;

  // 事件監聽器
  private listeners: Map<string, Array<(data?: any) => void>> = new Map();

  /**
   * 初始化手勢管理器
   */
  constructor(element: HTMLElement) {
    this.setupGestureDetection(element);
  }

  /**
   * 設置手勢檢測
   */
  private setupGestureDetection(element: HTMLElement): void {
    // 觸控事件
    element.addEventListener('touchstart', this.handleTouchStart.bind(this), { passive: true });
    element.addEventListener('touchmove', this.handleTouchMove.bind(this), { passive: false });
    element.addEventListener('touchend', this.handleTouchEnd.bind(this), { passive: true });
    element.addEventListener('touchcancel', this.handleTouchCancel.bind(this), { passive: true });

    // 防止瀏覽器默認行為
    element.addEventListener('touchmove', (e) => {
      if (e.touches.length > 1) {
        e.preventDefault(); // 防止雙指縮放時的頁面滾動
      }
    }, { passive: false });
  }

  /**
   * 處理觸控開始
   */
  private handleTouchStart(e: TouchEvent): void {
    this.touchStartTime = Date.now();

    if (e.touches.length === 1) {
      // 單指觸控
      this.touchStartX = e.touches[0].clientX;
      this.touchStartY = e.touches[0].clientY;

      // 設置長按定時器
      this.longPressTimer = setTimeout(() => {
        this.triggerGesture(GestureManager.GESTURE_TYPES.LONG_PRESS, {
          x: this.touchStartX,
          y: this.touchStartY
        });
      }, this.config.longPressThreshold);

    } else if (e.touches.length === 2) {
      // 雙指觸控（捏合）
      const dx = e.touches[0].clientX - e.touches[1].clientX;
      const dy = e.touches[0].clientY - e.touches[1].clientY;
      this.initialDistance = Math.sqrt(dx * dx + dy * dy);
    }
  }

  /**
   * 處理觸控移動
   */
  private handleTouchMove(e: TouchEvent): void {
    // 取消長按定時器
    if (this.longPressTimer) {
      clearTimeout(this.longPressTimer);
      this.longPressTimer = null;
    }

    if (e.touches.length === 2) {
      // 處理捏合手勢
      this.handlePinchGesture(e);
    }
  }

  /**
   * 處理觸控結束
   */
  private handleTouchEnd(e: TouchEvent): void {
    // 取消長按定時器
    if (this.longPressTimer) {
      clearTimeout(this.longPressTimer);
      this.longPressTimer = null;
    }

    const touchEndTime = Date.now();
    const duration = touchEndTime - this.touchStartTime;

    if (e.changedTouches.length === 1 && duration < this.config.tapThreshold) {
      // 點擊手勢
      const endX = e.changedTouches[0].clientX;
      const endY = e.changedTouches[0].clientY;
      const deltaX = Math.abs(endX - this.touchStartX);
      const deltaY = Math.abs(endY - this.touchStartY);

      // 檢查是否為有效的點擊（移動距離小）
      if (deltaX < 10 && deltaY < 10) {
        const now = Date.now();
        const timeSinceLastTap = now - this.lastTapTime;

        if (timeSinceLastTap < this.config.doubleTapThreshold) {
          // 雙擊
          this.triggerGesture(GestureManager.GESTURE_TYPES.DOUBLE_TAP, {
            x: endX,
            y: endY
          });
          this.lastTapTime = 0;
        } else {
          // 單擊
          this.triggerGesture(GestureManager.GESTURE_TYPES.TAP, {
            x: endX,
            y: endY
          });
          this.lastTapTime = now;
        }
      } else {
        // 滑動手勢
        this.handleSwipeGesture(endX, endY, duration);
      }
    }
  }

  /**
   * 處理滑動手勢
   */
  private handleSwipeGesture(endX: number, endY: number, duration: number): void {
    const deltaX = endX - this.touchStartX;
    const deltaY = endY - this.touchStartY;
    const velocityX = deltaX / duration;
    const velocityY = deltaY / duration;

    // 檢查是否達到滑動閾值
    if (Math.abs(deltaX) > this.config.swipeThreshold ||
        Math.abs(deltaY) > this.config.swipeThreshold) {

      let gestureType: string;

      if (Math.abs(deltaX) > Math.abs(deltaY)) {
        // 水平滑動
        gestureType = deltaX > 0
          ? GestureManager.GESTURE_TYPES.SWIPE_RIGHT
          : GestureManager.GESTURE_TYPES.SWIPE_LEFT;
      } else {
        // 垂直滑動
        gestureType = deltaY > 0
          ? GestureManager.GESTURE_TYPES.SWIPE_DOWN
          : GestureManager.GESTURE_TYPES.SWIPE_UP;
      }

      this.triggerGesture(gestureType, {
        deltaX,
        deltaY,
        velocityX,
        velocityY,
        duration
      });
    }
  }

  /**
   * 處理捏合手勢
   */
  private handlePinchGesture(e: TouchEvent): void {
    if (e.touches.length !== 2) return;

    const dx = e.touches[0].clientX - e.touches[1].clientX;
    const dy = e.touches[0].clientY - e.touches[1].clientY;
    const currentDistance = Math.sqrt(dx * dx + dy * dy);

    if (this.initialDistance > 0) {
      const scale = currentDistance / this.initialDistance;
      const scaleChange = Math.abs(1 - scale);

      if (scaleChange > this.config.pinchThreshold) {
        const gestureType = scale < 1
          ? GestureManager.GESTURE_TYPES.PINCH_IN
          : GestureManager.GESTURE_TYPES.PINCH_OUT;

        this.triggerGesture(gestureType, {
          scale,
          centerX: (e.touches[0].clientX + e.touches[1].clientX) / 2,
          centerY: (e.touches[0].clientY + e.touches[1].clientY) / 2
        });

        this.initialDistance = currentDistance; // 更新初始距離
      }
    }
  }

  /**
   * 觸發手勢事件
   */
  private triggerGesture(gestureType: string, data?: any): void {
    const listeners = this.listeners.get(gestureType) || [];
    listeners.forEach(listener => listener(data));

    // 同時觸發通用手勢事件
    const allListeners = this.listeners.get('*') || [];
    allListeners.forEach(listener => listener({ type: gestureType, ...data }));
  }

  /**
   * 註冊手勢監聽器
   */
  on(gestureType: string, callback: (data?: any) => void): () => void {
    if (!this.listeners.has(gestureType)) {
      this.listeners.set(gestureType, []);
    }

    const listeners = this.listeners.get(gestureType)!;
    listeners.push(callback);

    // 返回取消註冊函數
    return () => {
      const index = listeners.indexOf(callback);
      if (index > -1) {
        listeners.splice(index, 1);
      }
    };
  }

  /**
   * 銷毀手勢管理器
   */
  destroy(): void {
    this.listeners.clear();
    if (this.longPressTimer) {
      clearTimeout(this.longPressTimer);
    }
  }

  private longPressTimer: NodeJS.Timeout | null = null;
}
```

#### 卡片區域觸控優化 (`src/components/UI/BottomCardArea.tsx`)
```typescript
// 在BottomCardArea組件中添加觸控優化
const BottomCardArea: React.FC<BottomCardAreaProps> = (props) => {
  const containerRef = useRef<HTMLDivElement>(null);
  const [activeCardIndex, setActiveCardIndex] = useState<number | null>(null);

  // 初始化觸控優化
  useEffect(() => {
    if (!containerRef.current) return;

    // 優化所有卡片觸控目標
    const cards = containerRef.current.querySelectorAll('.card-item');
    cards.forEach(card => {
      TouchTargetOptimizer.optimizeElement(card as HTMLElement);
    });

    // 設置手勢管理器
    const gestureManager = new GestureManager(containerRef.current);

    // 註冊滑動手勢（切換卡片）
    const unsubscribeSwipe = gestureManager.on('swipe-left', () => {
      // 向左滑動：顯示更多卡片
      containerRef.current?.scrollBy({ left: 200, behavior: 'smooth' });
    });

    const unsubscribeSwipeRight = gestureManager.on('swipe-right', () => {
      // 向右滑動：返回之前的卡片
      containerRef.current?.scrollBy({ left: -200, behavior: 'smooth' });
    });

    return () => {
      unsubscribeSwipe();
      unsubscribeSwipeRight();
      gestureManager.destroy();
    };
  }, [props.gameState.hand]);

  // 處理卡片觸控
  const handleCardTouch = useCallback((cardId: string, index: number) =>
